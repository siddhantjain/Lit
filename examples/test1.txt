#program to find the decimal equivalent of a binary number given in the form of an array.

_binarytodecimal input_parameters [ int $arraysize] output_parameters [int $decimal];
    int $ans;
    int $count;
    int $multiply;
 
    $ans <- 0;
    $multiply <- 1;
    $count <- $arraysize;
    while($count >= 1)
        if ($a[$count] == 1)
            $ans <- $ans+ $multiply;
            $multiply <- $multiply*2;
            $count <- $count - 1;       #test comment inline
        else
            $multiply <- $multiply*2;
            $count <- $count - 1;
        endif
    endwhile

return [$ans];
end

	int[10] $a;
